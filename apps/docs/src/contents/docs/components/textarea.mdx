---
title: Textarea
description: Multiline text input with accessible styles.
---

## Install

```bash
catalix add textarea
```

## Usage

```tsx
import Textarea from "@/registry/react/core/textarea";

export default function Example() {
  return (
    <div className="space-y-2">
      <Textarea placeholder="Write something..." />
      <Textarea aria-invalid placeholder="Invalid" />
    </div>
  );
}
```
